001  (ns ring.ring-okta.session
002    (:require [clojure.data.codec.base64 :as b64]
003              [clojure.java.io :as io]
004              [clojure.string :as string]
005              [ring.util.response :as ring-response])
006    (:import (com.okta.saml SAMLValidator)))
007  
008  (defn- respond-to-okta-post [params okta-config]
009    (let [validator (SAMLValidator.)
010          config (.getConfiguration validator okta-config)
011          encoded-saml-response (String. (b64/decode (.getBytes (:SAMLResponse params) "UTF-8")))
012          saml-response (.getSAMLResponse validator encoded-saml-response config)
013          authenticated-user-email (.getUserID saml-response)
014          redirect-url (:RelayState params)]
015      {:redirect-url redirect-url
016       :authenticated-user-email (string/lower-case authenticated-user-email)}))
017  
018  (defn login [{:keys [params okta-config-location]}]
019    (let [okta-config (slurp (io/reader okta-config-location))
020          okta-response (respond-to-okta-post params okta-config)]
021      (-> (ring-response/redirect-after-post (:redirect-url okta-response))
022          (assoc-in [:session :okta/user] (:authenticated-user-email okta-response)))))
023  
024  (defn logout [{:keys [session]}]
025    (dissoc session :okta/user))
